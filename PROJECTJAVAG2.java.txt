		    private static double performCalculation(double num1, double num2, char operation) {
		        switch (operation) {
		            case '+': return num1 + num2;
		            case '-': return num1 - num2;
		            case '*': return num1 * num2;
		            case '/':
		                if (num2 == 0) {
		                    printError("\033[1;31mDIVISION BY ZERO!\033[0m");
		                    return Double.NaN; 
		                }
		                return num1 / num2;
		            default:
		                printError("\033[1;31mIVALID OPERATION!\033[0m");
		                return Double.NaN;
		        }
		    }
